## Step 1: Environment Setup

### Quick Setup with Environment Configuration

Our SDK includes a powerful environment configuration tool that makes setup effortless:

```bash

# Configure for Local Simulator (Recommended for Development)
./configure_environment.sh simulator
source .env.current

# Or configure for Live Visa Sandbox (Production Testing)

source .env.current
```

## Step 3: Configure Your Client

### TypeScript Configuration

```typescript
import { SecureHttpClient, PayoutBuilder } from 'visa-direct-sdk';

// Configure the HTTP client
const http = new SecureHttpClient({
  baseUrl: process.env.VISA_BASE_URL || 'http://127.0.0.1:8766',
  certPath: process.env.VISA_CERT_PATH || './certs/client.pem',
  keyPath: process.env.VISA_KEY_PATH || './certs/client.key',
  caPath: process.env.VISA_CA_PATH || './certs/ca.pem',
  jwe: {
    publicKey: process.env.VISA_JWE_PUBLIC_KEY || './certs/jwe_public.pem',
    privateKey: process.env.VISA_JWE_PRIVATE_KEY || './certs/jwe_private.pem'
  }
});

// Create the payout builder
const builder = PayoutBuilder.create(http);
```

### Python Configuration

```python
import os
from visa_direct_sdk.transport.secure_http_client import SecureHttpClient
from visa_direct_sdk.dx.builder import PayoutBuilder

# Configure the HTTP client
http = SecureHttpClient(
    base_url=os.getenv('VISA_BASE_URL', 'http://127.0.0.1:8766'),
    cert_path=os.getenv('VISA_CERT_PATH', './certs/client.pem'),
    key_path=os.getenv('VISA_KEY_PATH', './certs/client.key'),
    ca_path=os.getenv('VISA_CA_PATH', './certs/ca.pem'),
    jwe_public_key=os.getenv('VISA_JWE_PUBLIC_KEY', './certs/jwe_public.pem'),
    jwe_private_key=os.getenv('VISA_JWE_PRIVATE_KEY', './certs/jwe_private.pem')
)

# Create the payout builder
builder = PayoutBuilder.create(http)
```

## Step 4: Execute Your First Payout

### Simple FI to Card Transaction

<CodeGroup>

```typescript TypeScript
async function executeFirstPayout() {
  try {
    const receipt = await builder
      .forOriginator('fi-001')
      .withFundingInternal(true, 'LEDGER_CONFIRM_123')
      .toCardViaAlias({ 
        alias: '+447911123456', 
        aliasType: 'MOBILE' 
      })
      .forAmount('GBP', 5000) // £50.00
      .withIdempotencyKey('demo-first-payout')
      .execute();

    console.log('✅ Payout successful:', receipt);
    return receipt;
  } catch (error) {
    console.error('❌ Payout failed:', error);
    throw error;
  }
}

// Execute the payout
executeFirstPayout();
```

```python Python
async def execute_first_payout():
    try:
        receipt = builder \
            .for_originator('fi-001') \
            .with_funding_internal(True, 'LEDGER_CONFIRM_123') \
            .to_card_via_alias('+447911123456', 'MOBILE') \
            .for_amount('GBP', 5000) \
            .with_idempotency_key('demo-first-payout') \
            .execute()

        print('✅ Payout successful:', receipt)
        return receipt
    except Exception as error:
        print('❌ Payout failed:', error)
        raise error

# Execute the payout
await execute_first_payout()
```

## Step 7: Error Handling & Best Practices

### Comprehensive Error Handling

<CodeGroup>

```typescript TypeScript
import { 
  LedgerNotConfirmedError,
  CardValidationError,
  InsufficientFundsError,
  NetworkError,
  ValidationError
} from 'visa-direct-sdk';

async function robustPayout() {
  try {
    const receipt = await builder
      .forOriginator('fi-001')
      .withFundingInternal(true, 'LEDGER_CONFIRM_123')
      .toCardViaAlias({ alias: '+447911123456', aliasType: 'MOBILE' })
      .forAmount('GBP', 5000)
      .withIdempotencyKey('robust-demo')
      .execute();

    return receipt;
  } catch (error) {
    if (error instanceof LedgerNotConfirmedError) {
      console.error('Ledger confirmation failed:', error.message);
      // Handle ledger confirmation failure
    } else if (error instanceof CardValidationError) {
      console.error('Card validation failed:', error.message);
      // Handle invalid card details
    } else if (error instanceof InsufficientFundsError) {
      console.error('Insufficient funds:', error.message);
      // Handle insufficient funds
    } else if (error instanceof NetworkError) {
      console.error('Network error:', error.message);
      // Handle network issues
    } else if (error instanceof ValidationError) {
      console.error('Validation error:', error.message);
      // Handle validation failures
    } else {
      console.error('Unexpected error:', error);
    }
    throw error;
  }
}
```
